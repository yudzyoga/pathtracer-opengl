cmake_minimum_required(VERSION 3.10)
project(OpenGLPT)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(OpenGL REQUIRED)
find_package(GLEW REQUIRED)

# Add GLFW from ext/
add_subdirectory(ext/glfw)
add_subdirectory(ext/glm)
